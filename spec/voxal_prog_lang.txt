(define while cond clb
	(clb)
	(if (cond) 
		(while cond clb)))

(define main a1 a2 a3
	(define s
		(if (< a1 a2)
			a1
			a2))

	(print s "\nHello, world")

	(while (< s 10)
		(set s (+ s 1))
		(print s)))

<program> ::= { <function> }
<function> ::= '(' 'define' <ident> { <ident> } { <function> } { <call> } ')'
<call> ::= '(' <ident> { <param> } ')'
<param> ::= <ident> | <literal>
<literal> ::= '\".*\"' | <number>
<ident> ::= [a-zA-Z0-9\-]
<number> ::= [0-9]+
